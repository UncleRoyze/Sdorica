def SelectChallengeStage():
    def GetChallengePageAndCount(stage, sub_stage):
        page, count = 0, 0
        if stage in range(1, 6):
            page = 1
            if sub_stage in range(1, 5):
                count = (stage - 1) * 4 + sub_stage - 1
        elif stage in range(6, 14):
            page = ((stage - 6) / 4) + 2
            count = ((stage - 6) % 4) * 5 + sub_stage - 1
        elif stage == 14:
            page = 4
            count = sub_stage - 1
        return page, count
